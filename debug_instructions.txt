on gdb:
    set environment ASAN_OPTIONS=abort_on_error=1:print_stacktrace=1:symbolize=1:verbosity=1:log_threads=1


Options
detect_leaks=1	Enables leak detection (uses LeakSanitizer).
detect_stack_use_after_return=1	Catches stack-use-after-return errors (slower but powerful).
malloc_context_size=20	Number of stack frames to capture per allocation.
abort_on_error=1
print_stacktrace=1
symbolize=1
verbosity=0
log_threads=1

    mpirun.mpich build/driver/driver
    
    ASAN_OPTIONS="detect_leaks=1:symbolize=1:log_threads=1:log_path=asan.log" \

    mpirun.mpich build/driver/driver
