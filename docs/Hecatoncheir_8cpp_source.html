152 static DB_STATUS spawnWorkers(int num_procs, const std::vector<std::string> &hosts) {
153  DB_STATUS ret = DBERR_OK;
154  int rank, size;
155  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
156 
157  if (rank == 0) {
158  // spawning N+1 processes in total
159  int total_processes = num_procs + 1;
160  std::vector<char*> cmdsVec(total_processes, (char*) WORKER_EXECUTABLE_PATH.c_str());
161  char** cmds = cmdsVec.data();
162  int* error_codes = (int*)malloc(total_processes * sizeof(int));
163  MPI_Info* info = (MPI_Info*)malloc(total_processes * sizeof(MPI_Info));
164  int* np = (int*)malloc(total_processes * sizeof(int));
165 
166  // set spawn info for first two processes (ranks 0 and 1) on first host
167  for (int i = 0; i < total_processes; i++) {
168  if (i < 2) {
169  np[i] = 1;
170  MPI_Info_create(&info[i]);
171  MPI_Info_set(info[i], "host", hosts[0].c_str());
172  } else {
173  np[i] = 1;
174  MPI_Info_create(&info[i]);
175  // use host i-1 since there are N+1 processes for N workers
176  MPI_Info_set(info[i], "host", hosts[i-1].c_str());
177  }
178  }
179 
180  // Spawn total_processes workers
181  MPI_Comm_spawn_multiple(total_processes, cmds, NULL, np, info, 0, MPI_COMM_WORLD, &g_global_inter_comm, error_codes);
182 
183  for (int i = 0; i < total_processes; ++i) {
184  if (error_codes[i] != MPI_SUCCESS) {
185  logger::log_error(DBERR_MPI_INIT_FAILED, "Failed while spawning process " + std::to_string(i));
186  return DBERR_MPI_INIT_FAILED;
187  }
188  }
189 
190  // Cleanup
191  for (int i = 0; i < total_processes; i++) {
192  MPI_Info_free(&info[i]);
193  }
194  free(info);
195  free(np);
196  free(error_codes);
197 
198  // Rest of your existing code remains the same...
199  MPI_Intercomm_merge(g_global_inter_comm, 0, &g_global_intra_comm);
200  MPI_Comm_split(g_global_intra_comm, MPI_UNDEFINED, 0, &g_worker_comm);
201  MPI_Comm_rank(g_global_intra_comm, &rank);
202  MPI_Comm_size(g_global_intra_comm, &size);
203  g_global_rank = rank;
204  g_world_size = size;
205  g_workers_size = g_world_size-1;
206 
207  MPI_Barrier(g_global_intra_comm);
208 
209  ret = waitForResponse();
210  if (ret != DBERR_OK) {
211  logger::log_error(ret, "System initialization failed.");
212  return ret;
213  }
214 
215  MPI_Comm_free(&g_global_inter_comm);
