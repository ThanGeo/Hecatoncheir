682 void GeometryBatch::deserialize(const char *buffer, int bufferSize) {
683  size_t recID;
684  int vertexCount, partitionCount;
685  const char *localBuffer = buffer;
686 
687  // get object count
688  size_t objectCount = *reinterpret_cast<const size_t*>(localBuffer);
689  localBuffer += sizeof(size_t);
690 
691  // extend reserve space
692  geometries.reserve(geometries.size() + objectCount);
693 
694  // deserialize fields for each object in the batch
695  for (size_t i=0; i<objectCount; i++) {
696  // rec id
697  recID = *reinterpret_cast<const size_t*>(localBuffer);
698  localBuffer += sizeof(size_t);
699  // partition count
700  partitionCount = *reinterpret_cast<const int*>(localBuffer);
701  localBuffer += sizeof(int);
702  // partitions
703  std::vector<int> partitions(partitionCount * 2);
704  const int* partitionPtr = reinterpret_cast<const int*>(localBuffer);
705  for (size_t j = 0; j < partitionCount*2; j++) {
706  partitions[j] = partitionPtr[j];
707  }
708  localBuffer += partitionCount * 2 * sizeof(int);
709  // vertex count
710  vertexCount = *reinterpret_cast<const int*>(localBuffer);
711  localBuffer += sizeof(int);
712  // vertices
713  std::vector<double> coords(vertexCount * 2);
714  const double* vertexDataPtr = reinterpret_cast<const double*>(localBuffer);
715  for (size_t j = 0; j < vertexCount * 2; j++) {
716  coords[j] = vertexDataPtr[j];
717  }
718  localBuffer += vertexCount * 2 * sizeof(double);
719 
720  // add to batch
721  Geometry geometry(recID, partitions, vertexCount, coords);
722  this->addGeometryToBatch(geometry);
723  }
