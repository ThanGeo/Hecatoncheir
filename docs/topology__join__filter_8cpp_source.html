17  if (ret != DBERR_OK) {

38  static inline DB_STATUS relateMBRs(Shape* objR, Shape* objS, hec::QResultBase* queryResult) {
39  DB_STATUS ret = DBERR_OK;
40  // compute deltas
41  double d_xmin = objR->mbr.pMin.x - objS->mbr.pMin.x;
42  double d_ymin = objR->mbr.pMin.y - objS->mbr.pMin.y;
43  double d_xmax = objR->mbr.pMax.x - objS->mbr.pMax.x;
44  double d_ymax = objR->mbr.pMax.y - objS->mbr.pMax.y;
45 
46  // check for equality using an error margin because doubles
47  if (abs(d_xmin) < EPS) {
48  if (abs(d_xmax) < EPS) {
49  if (abs(d_ymin) < EPS) {
50  if (abs(d_ymax) < EPS) {
51  // equal MBRs
52  ret = forwardPair(objR, objS, MBR_EQUAL, queryResult);
53  if (ret != DBERR_OK) {
54  logger::log_error(ret, "Forward for equal MBRs stopped with error.");
55  }
56  return ret;
57  }
58  }
59  }
60  }
61  // not equal MBRs, check other relations
62  if (d_xmin <= 0) {
63  if (d_xmax >= 0) {
64  if (d_ymin <= 0) {
65  if (d_ymax >= 0) {
66  // MBR(s) inside MBR(r)
67  ret = forwardPair(objR, objS, MBR_S_IN_R, queryResult);
68  if (ret != DBERR_OK) {
69  logger::log_error(ret, "Forward for MBR(s) inside MBR(r) stopped with error.");
70  }
71  return ret;
72  }
73  } else {
74  if (d_ymax < 0 && d_xmax > 0 && d_xmin < 0 && d_ymin < 0) {
75  // MBRs cross each other
76  ret = forwardPair(objR, objS, MBR_CROSS, queryResult);
77  if (ret != DBERR_OK) {
78  logger::log_error(ret, "Forward for MBRs cross stopped with error.");
79  }
80  return ret;
81  }
82  }
83  }
84  }
85  if (d_xmin >= 0) {
86  if (d_xmax <= 0) {
87  if (d_ymin >= 0) {
88  if (d_ymax <= 0) {
89  // MBR(r) inside MBR(s)
90  ret = forwardPair(objR, objS, MBR_R_IN_S, queryResult);
91  if (ret != DBERR_OK) {
92  logger::log_error(ret, "Forward for MBR(r) inside MBR(s) stopped with error.");
93  }
94  return ret;
95  }
96  } else {
97  if (d_ymax > 0 && d_xmax < 0 && d_xmin > 0 && d_ymin > 0) {
98  // MBRs cross each other
99  ret = forwardPair(objR, objS, MBR_CROSS, queryResult);
100  if (ret != DBERR_OK) {
101  logger::log_error(ret, "Forward for MBRs cross stopped with error.");
102  }
103  return ret;
104  }
105  }
106  }
107  }
108  // MBRs intersect generally
109  ret = forwardPair(objR, objS, MBR_INTERSECT, queryResult);
110  if (ret != DBERR_OK) {
111  logger::log_error(ret, "Forward for MBRs intersect stopped with error.");
112  }
113  return ret;
114  }
