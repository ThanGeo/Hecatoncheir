232  DB_STATUS unpackSystemMetadata(SerializedMsg<char> &sysMetadataMsg) {
233  PartitioningTypeE partitioningType;
234  int partPartitionsPerDim, distPartitionsPerDim;
235  char *localBuffer = sysMetadataMsg.data;
236  // get system setup type
237  g_config.options.setupType = *reinterpret_cast<const SystemSetupTypeE*>(localBuffer);
238  localBuffer += sizeof(SystemSetupTypeE);
239  // get dist+part partitions per dimension
240  distPartitionsPerDim = *reinterpret_cast<const int*>(localBuffer);
241  localBuffer += sizeof(int);
242  partPartitionsPerDim = *reinterpret_cast<const int*>(localBuffer);
243  localBuffer += sizeof(int);
244  // get dist+part partitioning type
245  partitioningType = *reinterpret_cast<const PartitioningTypeE*>(localBuffer);
246  localBuffer += sizeof(PartitioningTypeE);
247  // set partitioning method
248  if (partitioningType == PARTITIONING_ROUND_ROBIN) {
249  // batch size to zero, since its irrelevant
250  g_config.partitioningMethod = new RoundRobinPartitioning(partitioningType, 0, partPartitionsPerDim);
251  } else if (partitioningType == PARTITIONING_TWO_GRID) {
252  // batch size to zero and distribution ppd to zero, both irrelevant to worker
253  g_config.partitioningMethod = new TwoGridPartitioning(partitioningType, 0, distPartitionsPerDim, partPartitionsPerDim);
254  } else {
255  logger::log_error(DBERR_INVALID_PARAMETER, "Unknown partitioning type while unpacking system metadata:", partitioningType);
256  return DBERR_INVALID_PARAMETER;
257  }
258  // get datapath length + string
259  // and set the paths
260  int length;
261  length = *reinterpret_cast<const int*>(localBuffer);
262  localBuffer += sizeof(int);
263  std::string datapath(localBuffer, localBuffer + length);
264  localBuffer += length * sizeof(char);
265  g_config.dirPaths.setNodeDataDirectories(datapath);
266  // pipeline
267  g_config.queryMetadata.MBRFilter = *reinterpret_cast<const int*>(localBuffer);
268  localBuffer += sizeof(int);
269  g_config.queryMetadata.IntermediateFilter = *reinterpret_cast<const int*>(localBuffer);
270  localBuffer += sizeof(int);
271  g_config.queryMetadata.Refinement = *reinterpret_cast<const int*>(localBuffer);
272  localBuffer += sizeof(int);
273 
274  return DBERR_OK;
