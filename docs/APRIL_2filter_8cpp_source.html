189  DB_STATUS IntermediateFilterEntrypoint(Shape* objR, Shape* objS, MBRRelationCase mbrRelationCase, hec::QResultBase* queryResult) {
190  DB_STATUS ret = DBERR_OK;
191  // switch based on how the MBRs intersect, to the appropriate intermediate filter
192  switch (mbrRelationCase) {
193  case MBR_R_IN_S:
194  ret = specializedTopologyRinSContainment(objR, objS, queryResult);
195  if (ret != DBERR_OK) {
196  logger::log_error(ret, "specialized topology R in S containment filter failed.");
197  }
198  break;
199  case MBR_S_IN_R:
200  ret = specializedTopologySinRContainment(objR, objS, queryResult);
201  if (ret != DBERR_OK) {
202  logger::log_error(ret, "specialized topology S in R containment filter failed.");
203  }
204  break;
205  case MBR_EQUAL:
206  ret = specializedTopologyEqual(objR, objS, queryResult);
207  if (ret != DBERR_OK) {
208  logger::log_error(ret, "specialized topology S in R containment filter failed.");
209  }
210  break;
211  case MBR_INTERSECT:
212  ret = specializedTopologyIntersection(objR, objS, queryResult);
213  if (ret != DBERR_OK) {
214  logger::log_error(ret, "specialized topology S in R containment filter failed.");
215  }
216  break;
217  default:
218  // not supported/unknown
219  logger::log_error(DBERR_INVALID_PARAMETER, "Unknown MBR relation case:", mbrRelationCase);
220  return DBERR_INVALID_PARAMETER;
221  }
222  return ret;
