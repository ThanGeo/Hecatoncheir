103  static DB_STATUS loadIntervals(std::string &intervalFilePath, Dataset &dataset, bool ALL) {
104  DB_STATUS ret = DBERR_OK;
105  int buf[3];
106  size_t objectCount;
107  size_t recID;
108  int sectionID, numIntervals, totalValues;
109  std::vector<uint32_t> intervals;
110  // open interval file
111  FILE* pFile = fopen(intervalFilePath.c_str(), "rb");
112  if (pFile == NULL) {
113  logger::log_error(DBERR_MISSING_FILE, "Could not open partitioned dataset file from path:", intervalFilePath);
114  return DBERR_MISSING_FILE;
115  }
116  // read object count
117  size_t elementsRead = fread(&objectCount, sizeof(size_t), 1, pFile);
118  if (elementsRead != 1) {
119  logger::log_error(DBERR_DISK_READ_FAILED, "Couldn't read object count");
120  ret = DBERR_DISK_READ_FAILED;
121  goto CLOSE_AND_EXIT;
122  }
123  // read APRIL for each object
124  for (size_t i=0; i<objectCount; i++) {
125  elementsRead = fread(&recID, sizeof(size_t), 1, pFile);
126  if (elementsRead != 1) {
127  logger::log_error(DBERR_DISK_READ_FAILED, "Couldn't read the recID","for object number",i,"of",objectCount);
128  ret = DBERR_DISK_READ_FAILED;
129  goto CLOSE_AND_EXIT;
130  }
131  elementsRead = fread(&buf, sizeof(int), 2, pFile);
132  if (elementsRead != 2) {
133  logger::log_error(DBERR_DISK_READ_FAILED, "Read", elementsRead,"instead of",2,"for object number",i,"of",objectCount);
134  ret = DBERR_DISK_READ_FAILED;
135  goto CLOSE_AND_EXIT;
136  }
137  sectionID = buf[0];
138  numIntervals = buf[1];
139  // X intervals are comprised of X*2 numbers [start,end)
140  totalValues = numIntervals*2;
141  intervals.resize(totalValues);
142  elementsRead = fread(&intervals.data()[0], sizeof(uint32_t), totalValues, pFile);
143  if (elementsRead != totalValues) {
144  logger::log_error(DBERR_DISK_READ_FAILED, "Couldn't read the intervals");
145  ret = DBERR_DISK_READ_FAILED;
146  goto CLOSE_AND_EXIT;
147 
148  }
149  // store
150  if (ALL) {
151  // add object to section map first
152  dataset.addObjectToSectionMap(sectionID, recID);
153  }
154  // add intervals
155  dataset.addIntervalsToAprilData(sectionID, recID, numIntervals, intervals, ALL);
156  }
157 
158 CLOSE_AND_EXIT:
159  fclose(pFile);
160  return ret;
161  }
