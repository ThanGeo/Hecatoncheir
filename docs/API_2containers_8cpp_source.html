15  Query* Query::createFromBuffer(char*& buffer) {
16  char* start = buffer;
17  // Read query type to determine subclass
18  int queryID = *reinterpret_cast<const int*>(buffer);
19  buffer += sizeof(int);
20 
21  QueryType type = *reinterpret_cast<const QueryType*>(buffer);
22  buffer += sizeof(QueryType);
23 
24  // Restore buffer for the right subclass to read fully
25  buffer = start;
26 
27  Query* query = nullptr;
28 
29  switch (type) {
30  case Q_RANGE:
31  query = new RangeQuery(); // empty constructor
32  break;
33  case Q_KNN:
34  query = new KNNQuery();
35  break;
36  case Q_DISTANCE_JOIN:
37  query = new DistanceJoinQuery();
38  break;
39  case Q_INTERSECTION_JOIN:
40  case Q_INSIDE_JOIN:
41  case Q_DISJOINT_JOIN:
42  case Q_CONTAINS_JOIN:
43  case Q_COVERS_JOIN:
44  case Q_COVERED_BY_JOIN:
45  case Q_MEET_JOIN:
46  case Q_EQUAL_JOIN:
47  case Q_FIND_RELATION_JOIN:
48  query = new PredicateJoinQuery();
49  break;
50  default:
51  logger::log_error(DBERR_QUERY_INVALID_TYPE, "Invalid query type in message. Type:", type);
52  return nullptr;
53  }
54 
55  // Deserialize into the new object
56  if (query->deserialize(buffer) < 0) {
57  delete query;
58  return nullptr;
59  }
60 
61  return query;
62  }
