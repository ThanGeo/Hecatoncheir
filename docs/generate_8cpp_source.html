64  }

241  M[(uint32_t)x1-rasterData.minCellX][(uint32_t)y1-rasterData.minCellY] = PARTIAL;
242  if(tMaxX < tMaxY){
243  x1 = x1 + stepX;
244  tMaxX = tMaxX + tDeltaX;
245  }else{

248  }
249  }
250  }
251  }

351  current_partial_cell++;
352  }

356  allIntervals.emplace_back(*(current_partial_cell-1) + 1);

360  aprilData.numIntervalsALL = allIntervals.size() / 2;
