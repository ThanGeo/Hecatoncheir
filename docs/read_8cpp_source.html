55  return DBERR_DISK_READ_FAILED;
56  }
57  elementsRead = fread(&partitionCount, sizeof(int), 1, pFile);
58  if (elementsRead != 1) {
59  logger::log_error(DBERR_DISK_READ_FAILED, "Couldn't read the partitionCount");
60  return DBERR_DISK_READ_FAILED;
61  }
62  // read partitions data
63  std::vector<int> partitionVector;
64  partitionVector.resize(partitionCount * 2);
65  elementsRead = fread(partitionVector.data(), sizeof(int), partitionCount * 2, pFile);

68  return DBERR_DISK_READ_FAILED;

70  // store partitions info
71  for (int i=0; i<partitionCount * 2; i+=2) {

73  }
74  // read vertex count
75  int vertexCount;
76  elementsRead = fread(&vertexCount, sizeof(int), 1, pFile);
77  if (elementsRead != 1) {
78  logger::log_error(DBERR_DISK_READ_FAILED, "Couldn't read the vertex count");
79  return DBERR_DISK_READ_FAILED;
80  }
81  // read the points
82  std::vector<double> coords(vertexCount*2);
83  elementsRead = fread(coords.data(), sizeof(double), vertexCount*2, pFile);
84  if (elementsRead != vertexCount*2) {
85  logger::log_error(DBERR_DISK_READ_FAILED, "Couldn't read the vertices");
86  return DBERR_DISK_READ_FAILED;
87  }
88  // reset mbr
89  object.mbr.pMin.x = std::numeric_limits<int>::max();
90  object.mbr.pMin.y = std::numeric_limits<int>::max();
91  object.mbr.pMax.x = -std::numeric_limits<int>::max();
92  object.mbr.pMax.y = -std::numeric_limits<int>::max();
93  for (auto it = coords.begin(); it != coords.end(); it += 2) {
94  // mbr
95  object.mbr.pMin.x = std::min(object.mbr.pMin.x, *it);
96  object.mbr.pMin.y = std::min(object.mbr.pMin.y, *(it+1));
97  object.mbr.pMax.x = std::max(object.mbr.pMax.x, *it);
98  object.mbr.pMax.y = std::max(object.mbr.pMax.y, *(it+1));
99  // vector data
100  object.addPoint(*it, *(it+1));
101  }
102  // correct
103  object.correctGeometry();
104 
105  return DBERR_OK;
106  }
